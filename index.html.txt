        // Toggle control panel
        function toggleControls() {
            const content = document.getElementById('controlContent');
            const icon = document.getElementById('toggleIcon');
            content.classList.toggle('collapsed');
            icon.classList.toggle('collapsed');
        }

        // Update image
        function updateImage(sectionNum) {
            const fileInput = document.getElementById('image' + sectionNum);
            const file = fileInput.files[0];
            
            if (file) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    const img = document.getElementById('displayImage' + sectionNum);
                    const placeholder = document.getElementById('placeholder' + sectionNum);
                    
                    img.src = e.target.result;
                    img.classList.add('visible');
                    placeholder.style.display = 'none';
                };
                reader.readAsDataURL(file);
            }
        }

        // Update all messages
        function updateAllMessages() {

            const subtitle = document.getElementById('subtitle').value;
            
            if (mainTitle.trim()) {
                document.getElementById('displayMainTitle').textContent = 'üíï ' + mainTitle + ' üíï';

            if (subtitle.trim()) {
                document.getElementById('displaySubtitle').textContent = subtitle;

            // Update section messages
            for (let i = 1; i <= 5; i++) {
                const messageInput = document.getElementById('message' + i);
                const displayMessage = document.getElementById('displayMessage' + i);
                
                if (messageInput.value.trim()) {
                    displayMessage.textContent = messageInput.value;
                }
            }
            
            // Update final section
            const finalTitle = document.getElementById('finalTitle').value;
            const finalMessage = document.getElementById('finalMessage').value;
            
            if (finalTitle.trim()) {
                document.getElementById('displayFinalTitle').textContent = 'üíù ' + finalTitle + ' üíù';
            }
            if (finalMessage.trim()) {
                document.getElementById('displayFinalMessage').innerHTML = finalMessage.replace(/\n/g, '<br>');
            }

        }
